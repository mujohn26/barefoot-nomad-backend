import tripsService from '../services/trips.services';
import response from '../helpers/response.helper';
 
/**
* Class for users to requested trips
*/
class tripRequest {
  /**
 * @param {Object} req req
 * @param {Object} res res
 * @param {Object} next ment
 * @returns {Object} hghfgjh
 */
  static async findrequest(req, res, next) {
    const managerId = req.user.id;
    const requests = await tripsService.findRequests(managerId);
 
    if (!requests[0]) {
      return response.errorMessage(res, 'no request found on your direct', 404, 'error');
    }
    req.requests = requests;
    next();
  }
 
  /**
 * @param {Object} req req
 * @param {Object} res res
 * @param {Object} next ment
 * @returns {Object} hghfgjh
 */
  static async approvalRequest(req, res, next) {
    const { tripId } = req.params;
    const managerId = req.user.id;
    const checkTrip = await tripsService.findTripRequest(tripId, managerId);
    if (!checkTrip[0]) { return response.errorMessage(res, 'no request with found', 404, 'error'); }
    if (checkTrip[0].status === 'approved') { return response.errorMessage(res, 'Already approved you can not approve it again', 401, 'error'); }
    req.tripFound = checkTrip[0];
    next();
  }
}
export default tripRequest;
